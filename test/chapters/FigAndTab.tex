    \label{tab:testtab}
    \begin{tabular}{ccc}
        \toprule
        OS           & TeX           & 测试情况 \\

        Windows 10   & TeX Live 2020 & ×        \\
        Ubuntu 20.04 & TeX Live 2021 & √        \\
        南大 TeX     & Overleaf      & √        \\
        \bottomrule
    \end{tabular}
\end{table}
